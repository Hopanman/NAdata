//1000 단위로 구분자를 추가하는 function.
function addCommas(str){
	   var arr,int,dec;
	   str += '';
	   arr = str.split('.');
	   int = arr[0] + '';
	   dec = arr.length>1?'.'+arr[1]:'';
	   return int.replace(/(\d)(?=(\d{3})+$)/g,"$1,") + dec;
}
//개봉전 평점 개선(선호도) Ajax function.
function queryAndUpdatePreference(mcode, preference) {
	
    var oAjax = new jindo.$Ajax("/movie/bi/mi/preferenceRegisterJson.nhn", {
    	onload : function (oRes) {
    		if (oRes.json().resultCode == "error") {						// 서버 오류
    			alert("서버 오류입니다.");
			} else if (oRes.json().resultCode == "readOnlyService") {	// 정기점검중
				alert("서비스 점검 중입니다.");
			} else if (oRes.json().resultCode == "login") {				// 로그인 유도
				if (confirm(CONSTANTS.sLoginMessage)) {
					parent.location.href = CONSTANTS.sLoginUrl + encodeURIComponent(parent.location);
				}
			} else {
	    		if(oRes.json().resultCode == 'success') {
	    			//기본 보기 화면 갱신.
					var percentage = oRes.json().interest / (oRes.json().interest + oRes.json().notInterested) * 100
					jindo.$Element("interest_cnt").html("<em class='blind'>보고싶어요</em>" + addCommas(oRes.json().interest));
					jindo.$Element("not_interest_cnt").html("<em class='blind'>글쎄요</em>" + addCommas(oRes.json().notInterested));
					jindo.$Element("preference_graph").css('width', percentage +"%");
					//넓게 보기 화면 갱신.
					jindo.$Element("interest_cnt_wide").html("<em class='blind'>보고싶어요</em>" + addCommas(oRes.json().interest));
					jindo.$Element("not_interest_cnt_wide").html("<em class='blind'>글쎄요</em>" + addCommas(oRes.json().notInterested));
					jindo.$Element("preference_graph_wide").css('width', percentage +"%");
					
					if(jindo.$Element("interest_cnt_point") != null && jindo.$Element("not_interest_cnt_point") != null && jindo.$Element("preference_graph_point") != null) {
						jindo.$Element("interest_cnt_point").html("<em class='blind'>보고싶어요</em>" + addCommas(oRes.json().interest));
						jindo.$Element("not_interest_cnt_point").html("<em class='blind'>글쎄요</em>" + addCommas(oRes.json().notInterested));
						jindo.$Element("preference_graph_point").css('width', percentage +"%");
					}
					
					if(jindo.$Element("interest_cnt_basic") != null && jindo.$Element("not_interest_cnt_basic") != null && jindo.$Element("preference_graph_basic") != null) {
						jindo.$Element("interest_cnt_basic").html("<em class='blind'>보고싶어요</em>" + addCommas(oRes.json().interest));
						jindo.$Element("not_interest_cnt_basic").html("<em class='blind'>글쎄요</em>" + addCommas(oRes.json().notInterested));
						jindo.$Element("preference_graph_basic").css('width', percentage +"%");
					}
					if(oRes.json().preferenceType == "interest") {
						alert("'보고싶어요' 가 선택되었습니다.");
					}
					else {
						alert("'글쎄요' 가 선택되었습니다.");
					}
	    		} else if(oRes.json().resultCode == 'CommonAlreadyExistInterestException') {
	    			alert("이미 '보고싶어요' 를 체크하셨습니다.");
	    		} else if(oRes.json().resultCode == 'CommonAlreadyExistNotInterestException') {
	    			alert("이미 '글쎄요' 를 체크하셨습니다.");
	    		} else if(oRes.json().resultCode == 'CommonAlreadyExistInterestNNBException') {
	    			alert("이미 '보고싶어요' 를 체크하셨습니다. \n기대평가는 영화별로 1인당 1회만 등록 가능합니다.");

	    		} else if(oRes.json().resultCode == 'CommonAlreadyExistInterestNDIException') {
	    			alert("이미 '보고싶어요' 를 체크하셨습니다. \n기대평가는 영화별로 1인당 1회만 등록 가능합니다.");
	    		} else if(oRes.json().resultCode == 'CommonAlreadyExistNotInterestNDIException') {
	    			alert("이미 '글쎄요' 를 체크하셨습니다. \n기대평가는 영화별로 1인당 1회만 등록 가능합니다.");
	    		}
	    		else {
	    			alert("서버 오류입니다.");
	    		}
			}
